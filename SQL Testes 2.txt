-- 1
SELECT ENAME, JOB, EMPNO
FROM 
 SCOTT.EMP
 
WHERE JOB = (SELECT JOB FROM SCOTT.EMP WHERE EMPNO = 7369)
-- 2
SELECT ENAME, JOB, EMPNO, SAL
FROM 
 SCOTT.EMP
 
WHERE JOB = (SELECT JOB FROM SCOTT.EMP WHERE EMPNO = 7369) AND SAL > ANY (SELECT SAL FROM SCOTT.EMP WHERE EMPNO = 7876);
-- 3
SELECT ENAME, JOB, SAL
FROM 
 SCOTT.EMP
 
WHERE SAL = (SELECT MIN(SAL) FROM SCOTT.EMP);
-- 4
SELECT DEPTNO, MIN(SAL) AS MIN_SAL
 
FROM SCOTT.EMP 
 
GROUP BY DEPTNO
 
HAVING MIN(SAL) > (SELECT MIN(SAL) FROM SCOTT.EMP WHERE DEPTNO = 20)
-- 5
SELECT P.ENAME, P.DEPTNO, P.JOB, D.LOC
 
FROM SCOTT.EMP P, SCOTT.DEPT D
 
WHERE D.LOC = 'DALLAS'
-- 6
SELECT E1.DEPTNO, E1.SAL
 
FROM SCOTT.EMP E1
WHERE SAL = (SELECT MIN(E2.SAL) FROM SCOTT.EMP E2 WHERE E1.DEPTNO = E2.DEPTNO)
-- 7
SELECT E1.EMPNO, E1.ENAME, E1.DEPTNO
 
FROM SCOTT.EMP E1
 
WHERE EXISTS ( SELECT 1 FROM SCOTT.EMP E2 WHERE E1.DEPTNO = E2.DEPTNO AND E2.ENAME LIKE '%T%')
-- 8
SELECT *
 
FROM SCOTT.EMP E1
 
WHERE E1.HIREDATE = (SELECT MAX(HIREDATE) FROM SCOTT.EMP E2 WHERE E1.DEPTNO = E2.DEPTNO)
ORDER BY E1.HIREDATE DESC
-- 9
SELECT E.ENAME, D.DNAME, D.LOC
 
FROM SCOTT.EMP E
 
JOIN SCOTT.DEPT D ON E.DEPTNO = D.DEPTNO
 
WHERE E.COMM IS NOT NULL
-- 10
SELECT E.DEPTNO, E.ENAME, D.DNAME, D.LOC
 
FROM SCOTT.EMP E
 
JOIN SCOTT.DEPT D ON E.DEPTNO = D.DEPTNO
 
WHERE E.ENAME = 'KING'
-- 11
SELECT *
 
FROM SCOTT.EMP E
 
JOIN SCOTT.DEPT D ON E.DEPTNO = D.DEPTNO
 
WHERE E.DEPTNO = 30
-- 12
SELECT ENAME, DEPTNO
 
FROM SCOTT.EMP 
 
WHERE ENAME LIKE '%A%'
-- 13
SELECT E.ENAME, E.JOB, E.EMPNO, D.DNAME
 
FROM SCOTT.EMP E, SCOTT.DEPT D
 
WHERE D.LOC = 'DALLAS'
-- 14
SELECT E.ENAME, E.JOB, E.EMPNO, D.DNAME, D.LOC
 
FROM SCOTT.EMP E, SCOTT.DEPT D
 
WHERE D.LOC = 'DALLAS' OR D.LOC = 'NEW YORK'
ORDER BY D.LOC
-- 15
SELECT E1.EMPNO AS "Código do Empregado", E1.ENAME AS "Nome do Empregado", 
 E2.EMPNO AS "Código do Gerente", E2.ENAME AS "Nome do Gerente"
 
FROM SCOTT.EMP E1
 
LEFT JOIN SCOTT.EMP E2 ON E1.MGR = E2.EMPNO
